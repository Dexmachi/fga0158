
\begin{multiplechoice}[rearrange={yes}, examcolumns=2]
  \begin{question}
     A definicao de niveis de acesso e o escopo de cada um dos elementos de uma determinada classe sao propriedades de Orientacao por Objeto trazidas pelo(a):

     \choice{Polimorfismo}
     \choice{Heranca}
     \choice[!]{Encapsulamento}
     \choice{Sobrescrita de metodos}

     \choice{Sobrecarga}
     \choice[!]{Sobrescrita}
     \choice{Inclusao}
     \choice{Nenhuma das alternativas anteriores}
  \end{question}

  \begin{question}

     \choice{protegidos}
     \choice[!]{estaticos}
  \end{question}

  \begin{question}
     Quando um metodo sofre diferentes implementacoes em uma classe, diz-se que esse metodo esta:

     \choice{estatico}
     \choice{dinamico}
  \end{question}
